chosen_exit = ""
while chosen_exit not in available_exits:
    chosen_exit = input("Please choose a direction: ")
    if chosen_exit.casefold()=="quit":
        print("Game over")
        break
    elif chosen_exit.casefold() in available_exits:
        print("aren't you glad you got out of there")
    else:
        continue
